\begin{frame}[c]{Klausur: Dos and Don'ts}{}
	\begin{columns}[c]
		\begin{column}{0.5\textwidth}
			\centering \textbf{Dos} \textcolor{TUMGreen}{\cmark}
			\vspace{\baselineskip}
			\begin{itemize}
				\item Student Card unbedingt mitbringen
				\item genügend Kugelschreiber mitnehmen
				\item Essen/Trinken mitnehmen
				\item ca. 20 Minuten vorher im Hörsaal sein
				\item Wörterbuch Deutsch $\leftrightarrow$ Muttersprache mitnehmen, falls benötigt: Keine Anmerkungen im Wörterbuch!
			\end{itemize}
		\end{column}
		\begin{column}{0.5\textwidth}
			\centering \textbf{Don'ts} \textcolor{TUMExtRed}{\xmark}
			\vspace{\baselineskip}
			\begin{itemize}
				\item rote oder grüne Stifte, Bleistift
				\item Tipp-Ex oder wärmeempfindliche Stifte
				\item Taschenrechner (zählt als Unterschleif!)
				\item eigene Schmierblätter, zusätzlicher Platz am Ende der Klausur
				\item irgendwelche Spickzettel -- wir finden die sowieso
			\end{itemize}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}[c]{Klausur: Vorbereitung}{}
	\begin{itemize}
		\item Übungsblätter und Altklausuren rechnen
		\item Altklausuren am besten ohne Hilfsmittel/Lösung und mit Stoppuhr $\rightarrow$ Gefühl für Klausurformat und eigene Stärken bekommen
		\item Assembly-Aufgaben $\rightarrow$ Möglichkeit, wenigstens Teilpunkte zu sammeln
		\item Überhangsklausur $\rightarrow$ zuerst Aufgaben machen, die ihr gut könnt
		\item Nicht verzweifeln, wenn ihr eine Aufgabe nicht auf Anhieb lösen könnt $\rightarrow$ überspringen und später zurückkehren
	\end{itemize}
\end{frame}
